git config 配置

配置一些常用 git 命令

vim ~/.gitconfig

[user]
    name = yourname  //如果name配置 github 帐号，会抓取你的 github 头像
    email = youremail@gmail.com
[push]
    default = simple
[alias]
    st = status
    co = checkout
    ci = commit
    br = branch
    di = diff
[color]
    ui = true
[core]
    quotepath = false
