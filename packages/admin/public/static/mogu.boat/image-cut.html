基本用法

视图:

<img id="J_CutImage" src="http://s18.mogucdn.com/b7/pic/140506/11jjem_kqzdm2smozbegq2ugfjeg5sckzsew_600x800.jpg_468x468.jpg">

引入：

<script type="text/javascript" src="jquery.crop.js"></script>

调用方法

/**
 * containerWidth，containerHeight。定义了容器宽高，图片会自动按比例缩放到容器宽高以内显示。
 * cropWidth，cropHeight。定义了裁剪区块宽高，初始化的时候会显示一块裁剪区块
 * aspectRatio，定义了拉伸比例，拉伸的时候会按比例绘制裁剪区块
 * initComplete，初始化完成回调方法
 */
$('#J_CutImage').crop({

    // 容器宽度
    containerWidth: 900,

    // 容器高度
    containerHeight: 600,

    // 初始化裁剪区块宽度
    cropWidth: 640,

    // 初始化裁剪区块高度
    cropHeight: 960,

    // 拉伸比例
    aspectRatio: 2/3,

    // 初始化完成回调
    initComplete: function(){
        alert('裁剪初始化完成');
    }
});
